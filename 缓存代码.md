async queryOrderItemGroup () {
      let params = {
        pageIndex: 1,
        pageSize: 99,
        mainOrderItemId: this.groupData.orderItemId || this.groupData.id,
      }
      const res = await this.$api({
                                    method: 'post',
                                    url: '/ts-bs-his/ts-bs-cln-out/queryOrderItemGroup',
                                    data: params,
                                    headers: {
                                      'Content-Type': 'application/json; charset=utf-8',
                                    },
                                  })
      let resArr = res.data.list || [];
      this.groupList = this.getDefaultData(resArr.filter(item => item.mainOrderItemId));
      const parentData=this.groupList = this.getDefaultData(resArr.filter(item => !item.mainOrderItemId));  
      let uuid=this.$utils.uuid(16)
      this.groupList.forEach((item) => {
        if(parentData[0]?.defaulSampleOrPositionCode){
          item.defaulSampleOrPositionCode=parentData[0].defaulSampleOrPositionCode
          item.defaultSampleOrPositionName=parentData[0].defaultSampleOrPositionName
        }
        item.packageRelationId=uuid
        
        if (item.openAllowDelete == "N") {
          this.$nextTick(() => {
            this.$refs.groupTable.toggleRowSelection(item, true);
          });
        } else {
          this.$nextTick(() => {
            this.$refs.groupTable.toggleRowSelection(item, false);
          });
        }
      });
     console.log("this.groupList",this.groupList)
    }
