# 基础配置(待完善)

```javascript
import { PivotSheet } from "@antv/s2";
//透视表
const s2 = new PivotSheet(container, dataCfg, s2Options);
s2.render();
const s2Options = {
      width: 600,height: 480,
      // 冻结行头(即行的标题),默认开启
      frozenRowHeader: true,
      //开启树状模式
      hierarchyType: 'tree',
};
```
