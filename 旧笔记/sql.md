SELECT a.id, a.dept_id, a.dept_name, a.DOCTOR_ID, a.DOCTOR_name, a.doc_level, a.doc_level_name, a.TIME_QUANTUM_name, a.order_date
FROM REG.REG_DOCTORS_SCHEDULING a
LEFT JOIN reg.reg_time_frame b
ON a.TIME_QUANTUM = b.TIME_QUANTUM
AND b.IS_DELETE = 'N'
AND b.ORG_CODE = 'BZK'
AND b.HOSP_CODE = 'BZK01'
AND b.USE_TYPE = '0'
AND TO_DATE(TO_CHAR(SYSDATE, 'HH24:MI'), 'HH24:MI')
BETWEEN TO_DATE(b.START_TIME, 'HH24:MI:SS') AND TO_DATE(b.END_TIME, 'HH24:MI:SS')
WHERE a.STOP_FLAG = '0'
AND a.IS_DELETE = 'N'
AND a.ORG_CODE = 'BZK'
AND a.HOSP_CODE = 'BZK01'
AND a.DOCTOR_NAME = '医生'
AND a.ORDER_DATE BETWEEN TRUNC(SYSDATE) AND (TRUNC(SYSDATE) + INTERVAL '23:59:59.999999' HOUR TO SECOND)
FETCH FIRST 1 ROWS ONLY;
