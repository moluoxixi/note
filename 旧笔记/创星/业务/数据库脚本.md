DECLARE

-- 声明变量（机构编码、机构名称）
v_org_code VARCHAR2(32);
v_org_name VARCHAR2(100);
v_hosp_code VARCHAR2(32);
v_id1 VARCHAR2(32) := '187040710347751424'; -- 初始化ID变量1
v_id2 VARCHAR2(32) := '64799079137896648096'; -- 初始化ID变量2
v_first_iteration BOOLEAN := TRUE; -- 新增变量，用于标记是否为第一次迭代

-- 声明游标
CURSOR c_org_data IS
SELECT ORG_CODE, ORG_NAME,PARENT_CODE AS HOSP_CODE FROM BASEDATA.BAS_ORG_ORGANIZATION b
WHERE b.ORG_TYPE = '1';
BEGIN

-- 打开游标
OPEN c_org_data;
-- 循环遍历游标结果
LOOP
-- 根据是否为第一次迭代决定使用固定ID还是生成新的UUID
IF v_first_iteration THEN
-- 第一次迭代，使用初始化的ID值
v_first_iteration := FALSE; -- 标记为非第一次迭代
ELSE
-- 非第一次迭代，生成新的UUID
v_id1 := LOWER(RAWTOHEX(SYS_GUID()));
v_id2 := LOWER(RAWTOHEX(SYS_GUID()));
END IF;
-- 获取下一行数据
FETCH c_org_data INTO v_org_code, v_org_name, v_hosp_code;
-- 检查是否还有数据
EXIT WHEN c_org_data%NOTFOUND;

-- 写入insert语句
INSERT INTO BASEDATA.BAS_DIAGNOSIS

END LOOP;

-- 关闭游标
CLOSE c_org_data;
COMMIT; -- 提交事务，确保插入操作生效
EXCEPTION
WHEN OTHERS THEN
ROLLBACK; -- 如果有异常，回滚事务
RAISE; -- 抛出异常
END;
